Plist

A property list (plist) file is a file with file extension .plist. It is created for the storage and retrieval of application data. Internally, it is structured and stored as an XML file. Externally, it displays key-value pairings as shown in the example below.

A plist file can be created to contain an array or a dictionary with key-value pairings. Dictionaries can be embedded into each other creating a complex data structure. For example, in the above plist, we have a top-level dictionary with key = stock symbol and value = company data dictionary. The company data dictionary has the key-value pairings as shown above for IBM.

Code Common to All Examples Below

import Foundation

import SwiftUI

 

/*

 Create an empty mutable dictionary using initializer () and store its unique

 id into the global variable myFavoritesDict_StockSymbol_CompanyDataDict

 where KEY=Stock Symbol, VALUE: Dictionary of Company Data

 */

var myFavoritesDict_StockSymbol_CompanyDataDict = Dictionary<String, Any>()

 

/*

 Create an empty mutable array using initializer () and store its unique

 id into the global variable orderedSearchableCompaniesList.

 */

var orderedSearchableCompaniesList = [String]()

 

/*

 Difference between fileprivate and private:

 fileprivate --> makes this constant accessible anywhere only inside this source file.

 private     --> makes it accessible only inside the type (e.g., class, struct) that declared it.

 */

fileprivate let documentDirectory = FileManager.default.urls(for: .documentDirectory, in: .userDomainMask)[0]

 

Read File from Document Directory into an Array

// Obtain the URL of the file in the document directory on the user's device

let urlOfFileInDocDir = documentDirectory.appendingPathComponent("OrderedSearchableCompaniesList")

       

// Instantiate an NSArray object and initialize it with the contents of the file

let arrayFromFile: NSArray? = NSArray(contentsOf: urlOfFileInDocDir)

       

if let arrayObtained = arrayFromFile {

     // Store the unique id of the created array into the global variable

     orderedSearchableCompaniesList = arrayObtained as! [String]

} else {

     fatalError("OrderedSearchableCompaniesList file is not found in the document directory on the user's device!")

}

 

Write an Array into a File in Document Directory

// Obtain the URL of the file in document directory on user's device

let urlOfFileInDocDirectory = documentDirectory.appendingPathComponent("OrderedSearchableCompaniesList")

 

/*

 Swift Array does not yet provide the 'write' function, but NSArray does.

 Therefore, typecast the Swift array as NSArray so that we can write it.

 */

(orderedSearchableCompaniesList as NSArray).write(to: urlOfFileInDocDirectory, atomically: true)

 

Read Plist File from Main Bundle into a Dictionary

// Obtain the URL of the CompaniesILike.plist file in the main bundle (project folder)

let urlOfPlistFileInMainBundle = Bundle.main.url(forResource: "CompaniesILike", withExtension: "plist")

 

// Instantiate an NSMutableDictionary object and initialize it with the contents of the plist file.

let dictionaryFromPlistFileInMainBundle: NSMutableDictionary? = NSMutableDictionary(contentsOf: urlOfPlistFileInMainBundle!)

 

if let dictionaryFromPlistFile = dictionaryFromPlistFileInMainBundle {

   

    // Store the unique id of the created dictionary into the global variable

    myFavoritesDict_StockSymbol_CompanyDataDict = dictionaryFromPlistFile as! [String : Any]

   

} else {

    fatalError("CompaniesILike.plist file is not found in the main bundle (project folder)!")

}

 

Read Plist File from Document Directory into a Dictionary

    // Obtain the URL of the plist file in the document directory on the user's device

    let urlOfPlistFileInDocumentDirectory = documentDirectory.appendingPathComponent("CompaniesILike.plist")

   

    /*

     NSMutableDictionary manages an *** unordered *** collection of mutable (modifiable) key-value pairs.

     Instantiate an NSMutableDictionary object and initialize it with the contents of the plist file.

     */

    let dictionaryFromFile: NSMutableDictionary? = NSMutableDictionary(contentsOf: urlOfPlistFileInDocumentDirectory)

   

    if let dictionaryFromFileInDocumentDirectory = dictionaryFromFile {

       

        // Store the unique id of the created dictionary into the global variable

        myFavoritesDict_StockSymbol_CompanyDataDict = dictionaryFromFileInDocumentDirectory as! [String : Any]

       

        /*

         If you write an NSDictionary or NSMutableDictionary to a file and read it back, you will

            get a completely different order of key-value pairings.

         */

    } else {

        fatalError("CompaniesILike.plist file is not found in the document directory on the user's device!")

    }

 

Write a Dictionary into a Plist File in Document Directory

    // Obtain the URL of the plist file in the document directory on the user's device

    let urlOfPlistFileInDocumentDirectory = documentDirectory.appendingPathComponent("CompaniesILike.plist")

 

    /*

     Swift Dictionary does not yet provide the 'write' function, but NSDictionary does.

     Therefore, typecast the Swift dictionary as NSDictionary so that we can write it.

     */

    let dictionary = myFavoritesDict_StockSymbol_CompanyDataDict as NSDictionary

   

    // Write the dictionary to the document directory on the user's device

    dictionary.write(to: urlOfPlistFileInDocumentDirectory, atomically: true)

   

    /*

     The flag "atomically" specifies whether the file should be written atomically or not.

    

     If flag atomically is TRUE, the dictionary is first written to an auxiliary file, and

     then the auxiliary file is renamed CompaniesILike.plist.

    

     If flag atomically is FALSE, the dictionary is written directly to CompaniesILike.plist.

     This is a bad idea since the file can be corrupted if the system crashes during writing.

    

     The TRUE option guarantees that the file will not be corrupted even if the system crashes during writing.

     */
